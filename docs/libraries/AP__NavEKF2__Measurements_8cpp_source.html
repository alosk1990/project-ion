581 // calculate filtered offset between GPS height measurement and EKF height estimate
582 // offset should be subtracted from GPS measurement to match filter estimate
583 // offset is used to switch reversion to GPS from alternate height data source
584 void NavEKF2_core::calcFiltGpsHgtOffset()
585 {
586  // Estimate the WGS-84 height of the EKF's origin using a Bayes filter
587 
588  // calculate the variance of our a-priori estimate of the ekf origin height
589  float deltaTime = constrain_float(0.001f * (imuDataDelayed.time_ms - lastOriginHgtTime_ms), 0.0f, 1.0f);
590  if (activeHgtSource == HGT_SOURCE_BARO) {
591  // Use the baro drift rate
592  const float baroDriftRate = 0.05f;
593  ekfOriginHgtVar += sq(baroDriftRate * deltaTime);
594  } else if (activeHgtSource == HGT_SOURCE_RNG) {
595  // use the worse case expected terrain gradient and vehicle horizontal speed
596  const float maxTerrGrad = 0.25f;
597  ekfOriginHgtVar += sq(maxTerrGrad * norm(stateStruct.velocity.x , stateStruct.velocity.y) * deltaTime);
598  } else if (activeHgtSource == HGT_SOURCE_GPS) {
599  // by definition we are using GPS height as the EKF datum in this mode
600  // so cannot run this filter
601  return;
602  }
603  lastOriginHgtTime_ms = imuDataDelayed.time_ms;
604 
605  // calculate the observation variance assuming EKF error relative to datum is independant of GPS observation error
606  // when not using GPS as height source
607  float originHgtObsVar = sq(gpsHgtAccuracy) + P[8][8];
608 
609  // calculate the correction gain
610  float gain = ekfOriginHgtVar / (ekfOriginHgtVar + originHgtObsVar);
611 
612  // calculate the innovation
613  float innovation = - stateStruct.position.z - gpsDataDelayed.hgt;
614 
615  // check the innovation variance ratio
616  float ratio = sq(innovation) / (ekfOriginHgtVar + originHgtObsVar);
617 
618  // correct the EKF origin and variance estimate if the innovation variance ratio is < 5-sigma
619  if (ratio < 5.0f) {
620  EKF_origin.alt -= (int)(100.0f * gain * innovation);
621  ekfOriginHgtVar -= fmaxf(gain * ekfOriginHgtVar , 0.0f);
622  }
623 }
